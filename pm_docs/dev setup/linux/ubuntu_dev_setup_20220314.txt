   # specify shared directory
   path = /home/share
   # allow writing
   writable = yes
   # allow guest user (nobody)
   guest ok = yes
   # looks all as guest user
   guest only = yes
   # set permission [777] when file created
   force create mode = 777
   # set permission [777] when folder created
   force directory mode = 777
